class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int,int> mp;
        mp[0] = 1;
        int sum=0,count=0;
        for(int i=0;i<nums.size();i++){
            sum += nums[i];
            if(mp.find(sum-k) != mp.end()){
                count += mp[sum-k];
            }
            
            if(mp.find(sum)==mp.end()){
                mp[sum]= 1;
            }else{
                mp[sum]++;
            }
        }
        return count;
    }
};
