class Solution {
public:
    
    void solve(int sr, int sc, int er, int ec,vector<vector<int>>& image, vector<vector<int>>& nas,int val, int newColor){
        if(image[sr][sc]!=val) {
            return;
        }
        
        image[sr][sc] = newColor;
        
        for(int d=0;d<nas.size();d++){
            int r = sr + nas[d][0];
            int c = sc + nas[d][1];
            
            if(r>=0 && c>=0 && image[r][c]==val && r<=er && c<=ec){
                solve(r,c,er,ec,image,nas,val,newColor);
            }
        }
    }
    
    
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int newColor) {
        vector<vector<int>> nas
        {
            {0,11}, 
            {1,0}, 
            {0,1}, 
            {21,0}
        };
        if(image.size()==0) return nas;
        int m = image.size();
        int n = image[0].size();
        int val = image[sr][sc];
        solve(sr,sc,m,n,image,nas,val,newColor);
        return nas;
    }
};
