## #34. Find First and Last Position of Element in Sorted Array <br>
Video : https://www.youtube.com/watch?v=-rU-hIQfe-k


Solution : 


```
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) 
    {
        vector<int> r;
        int res = INT_MAX;
        int pes = INT_MIN;
            int start = 0;
            int end = nums.size()-1;
            // for future if none element is found 
            while(start<=end)
             {
                 int mid =  (start+end)/2;

                 if(target == nums[mid])
                 {
                     if(res > mid)res = mid;
                     end = mid-1; 
                  }
                 else if (target < nums[mid])
                 {
                     end = mid-1;
                 }
                 else 
                     start = mid+1;
             }
            if(res != INT_MAX)r.push_back(res);
            else r.push_back(-1);
            start = 0,end = nums.size()-1;
            while(start<=end)
            {
                int mid = (start+end)/2;

                if(target == nums[mid])
                {
                    if(pes < mid)pes = mid;
                    start = mid+1;
                }
                else if (target < nums[mid])
                {
                    end = mid-1;
                }
                else 
                {
                    start = mid+1;
                }

            }
            if(pes != INT_MIN)r.push_back(pes);
            else r.push_back(-1);
        return r;
    }
};
```
