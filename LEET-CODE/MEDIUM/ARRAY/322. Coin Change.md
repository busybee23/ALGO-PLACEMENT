using namespace std;
class Solution {

    int min(int x,int y){
        if(x<y)
            return x;
        else
            return y;
    }
    
    public:
    int minex(vector<int>& coins,int sum,int i){
        if(sum==0)
            return 1;

        if(sum<=0 )
            return -1;
        
        // recusrive case
        return min(1 + minex(coins,sum-coins[i],i),minex(coins,sum,i+1));
        }
    
    int coinChange(vector<int>& coins, int sum) { 
        return (coins,sum,0);
        
    }
};
