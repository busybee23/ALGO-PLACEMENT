```
class Solution
{
  public:
    long long int count( int arr[], int len, int sum)
    {
       int dp[sum+1];
       dp[0]=1;
        //memset(dp,0,sizeof(dp));
       for(int i=0;i<len;i++){
           for(int j=arr[i];j<sum+1;j++){
               dp[j]+= dp[j-arr[i]];
           }
       }
        return dp[sum];
    }
};
```
