### Solution 1 : ( Accepted )

TIL : Count how many times each number appears. If a number appears n times, then n * (n â€“ 1) // 2 good pairs can be made with this number.

```
class Solution {
public:
    int numIdenticalPairs(vector<int>& nums) {
        int count = 0;
        unordered_map <int,int> mp;
        for(auto it:nums) mp[it]++;
        for(auto cheese : mp){
            count += (cheese.second*(cheese.second-1))/2;
        }
        return count;
    }
};

### Solution 2 : ( can be done using n2 too)

```
class Solution {
public:
    int numIdenticalPairs(vector<int>& nums) {
    int count = 0;
    for(int i=0;i<nums.size();i++){
        for(int j=i;j<nums.size();j++){
            if(nums[i]==nums[j] && i<j) count++;
        }
    }
        return count;
    }       
};
