class Solution {
public:
    int prefixCount(vector<string>& words, string pref) {
        vector<string> ans;
        int count=0; 
        int size_p = pref.size();
        int size = words.size();
        for(int i=0;i<size;i++){
            for(int j=0;j<words[i].size();j++) {
                bool flag = true;
                while(size_p){
                    if(words[j]!=pref[j]){
                        flag = false;
                        break;
                    }
                }
                if(flag==true){
                    count++;
                }
            }
        }
        return count;
        }
    }
};
