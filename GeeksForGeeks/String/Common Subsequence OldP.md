class Solution {
  public:
    int commonSubseq(string S1, string S2){
        vector<int> arr(26 , 0);
        for(auto it : S1){
            if(it>='A' && it<='Z') arr[it-'A']++;
        }
        for(auto it : S2){
            if(it>='A' && it<='Z' && arr[it-'A']) return 1; 
        }
        return 0;
    }
};
