class Solution{
public:
    vector<int> leftSmaller(int n, int a[]){
    vector<int> ans(n);
    stack<int> st;
    ans[0]=-1;
    st.push(a[0]);
    for(int i=1;i<n;i++){ 
        
    while(!st.empty()and st.top()>=a[i])
    {
    st.pop();
    }

    if(st.empty())
    {
    ans[i]=-1;
    }
    else
    {
    ans[i]=st.top();
    }

    st.push(a[i]);
    }
return ans;
}
};
