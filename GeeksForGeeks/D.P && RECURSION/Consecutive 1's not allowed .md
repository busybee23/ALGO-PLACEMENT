Solution : (Accepted on gfg)

```
class Solution{
public:
	// #define ll long long
	ll countStrings(int n) {
	    int ocz = 1;
	    int oco = 1;
	    
	    for(int i=2;i<=n;i++)
	    {
	        int ncz = oco%1000000007;
	        int nco = ocz + oco%1000000007;
	        
	        ocz = ncz%1000000007;
	        oco = nco%1000000007;
	        
	    }
	    
	    return (ocz+oco)%1000000007;
	}
};
```
