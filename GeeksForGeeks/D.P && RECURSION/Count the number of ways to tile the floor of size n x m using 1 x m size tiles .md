### SOLUTION : ( Accepted )

```
class Solution{
	public:
		int countWays(int n, int m)
	{
	    long long int dp[n+1] = {0};
	    for(int i=1;i<=n;i++){
	        if(i<m){
	            dp[i] = 1;
	        }else if(i==m){
	            dp[i] = 2;
	        }else{
	            dp[i] = (dp[i-1] + dp[i-m])%1000000007;
	        }
	    }
	    return dp[n];
	}
};
```
